27	class ArgumentParsingExit(Exception):
28	    def __init__(self, message: str, exit_code: int) -> None:
29	        super().__init__(message)
30	        self.message = message
31	        self.exit_code = exit_code

34	class RecordingArgumentParser(argparse.ArgumentParser):
35	    def __init__(self, *args: object, **kwargs: object) -> None:
36	        super().__init__(*args, **kwargs)
37	        self._captured_output: list[str] = []
38	
39	    def _print_message(self, message: str | None, file: object | None = None) -> None:
40	        if message:
41	            self._captured_output.append(message)
42	
43	    def _consume_output(self) -> str:
44	        merged = "".join(self._captured_output).strip()
45	        self._captured_output.clear()
46	        return merged
47	
48	    def exit(self, status: int = 0, message: str | None = None) -> None:
49	        text = self._consume_output()
50	        if message:
51	            text = f"{text}\n{message}".strip() if text else message.strip()
52	        if not text and status == 0:
53	            text = self.format_help().strip()
54	        raise ArgumentParsingExit(text, status)
55	
56	    def error(self, message: str) -> None:
57	        usage = self.format_usage().strip()
58	        self.exit(2, f"{usage}\n{self.prog}: error: {message}")

61	def _parse_on_off(value: str) -> bool:
62	    lowered = value.strip().lower()
63	    if lowered in {"on", "true", "1", "yes", "y"}:
64	        return True
65	    if lowered in {"off", "false", "0", "no", "n"}:
66	        return False
67	    raise argparse.ArgumentTypeError("请使用 on/off")

72	        root_override = os.environ.get("AI_ASSISTANT_ROOT")
73	        if root_override:

228	def _result_from_message(
229	    message: str,
230	    error_code: int = 1,
231	    data: dict[str, object] | None = None,
232	) -> CommandResult:
233	    ok = not message.startswith("❌")
234	    return CommandResult(ok, message, 0 if ok else error_code, data=data or {})

271	def _handle_context(args: argparse.Namespace, ctx: AppContext) -> CommandResult:
272	    if args.action in {"set", "add"}:
273	        try:
274	            if args.action == "set":
275	                return _result_from_message(ctx.context_service.set_context(args.file, args.start, args.end))
276	            return _result_from_message(ctx.context_service.add_context(args.file, args.start, args.end))
277	        except (FileNotFoundError, ValueError) as exc:
278	            return CommandResult(False, f"❌ {exc}", 1)
279	        except Exception as exc:
280	            return CommandResult(False, f"❌ 设置上下文失败：{exc}", 1)
281	    if args.action == "list":
282	        return _result_from_message(ctx.context_service.list_context())
283	    if args.action == "clear":
284	        return _result_from_message(ctx.context_service.clear_context())
285	    if args.action == "ask":
286	        try:
287	            prompt = ctx.context_service.build_prompt(args.question)
288	        except ValueError as exc:
289	            return CommandResult(False, f"❌ {exc}", 1)
290	
291	        ctx.history_service.trim_and_summarize(ctx.ai_gateway.summarize_messages)
292	        messages = ctx.history_service.build_messages_for_request(
293	            user_prompt=prompt,
294	            include_recent_history=True,
295	            include_recent_events=True,
296	            extra_system_messages=[build_cli_command_rules_prompt()],
297	        )
298	        response = ctx.ai_gateway.chat(
299	            messages,
300	            stream_override=None,
301	            print_stream=True,
302	            timeout=90,
303	            attempt_callback=RuntimeFeedback(enabled=True).as_attempt_callback(),
304	        )
305	        stream_enabled = ctx.config_service.get_active_profile().stream
306	        if response.ok:
307	            ctx.history_service.append_exchange(f"context ask: {args.question}", response.content)
308	            return CommandResult(
309	                True,
310	                response.content,
311	                data={"streamed_output": bool(stream_enabled)},
312	            )
313	
314	        ctx.history_service.append_exchange(f"context ask: {args.question}", response.content)
315	        return CommandResult(False, response.content, 1, data={"streamed_output": False})
316	
317	    return CommandResult(False, "❌ 未知 context 子命令", 2)

320	def _handle_backup(args: argparse.Namespace, ctx: AppContext) -> CommandResult:
321	    if args.action == "create":
322	        ok, message = ctx.backup_service.create_backup(args.file, args.keep)
323	        return CommandResult(ok, message, 0 if ok else 1)
324	    if args.action == "status":
325	        return CommandResult(True, ctx.backup_service.backup_status(args.file))
326	    if args.action == "list":
327	        backups = ctx.backup_service.list_backups(args.file)
328	        if not backups:
329	            return CommandResult(True, f"📝 文件没有备份：{Path(args.file).expanduser().resolve()}")
330	        lines = [f"📝 备份列表：共 {len(backups)} 个"]
331	        for index, item in enumerate(backups, 1):
332	            lines.append(f"  {index}. {item.get('backup_file')} ({item.get('ts', '')})")
333	        return CommandResult(True, "\n".join(lines))
334	    if args.action == "restore":
335	        ok, message = ctx.backup_service.restore_backup(
336	            args.backup_file,
337	            target_path=args.target,
338	            backup_target_before_restore=not args.no_backup_target_before_restore,
339	        )
340	        return CommandResult(ok, message, 0 if ok else 1)
341	    if args.action == "clean":
342	        ok, message = ctx.backup_service.clean_backups(args.file, args.keep)
343	        return CommandResult(ok, message, 0 if ok else 1)
344	    return CommandResult(False, "❌ 未知 backup 子命令", 2)

347	def _handle_config(args: argparse.Namespace, ctx: AppContext) -> CommandResult:
348	    service = ctx.config_service
349	    if args.action == "add":
350	        ok, message = service.add_profile(
351	            profile_id=args.profile,
352	            name=args.name,
353	            api_key=args.api_key,
354	            api_url=args.api_url,
355	            model=args.model,
356	            stream=args.stream,
357	            overwrite=False,
358	        )
359	        return CommandResult(ok, message, 0 if ok else 1)
360	    if args.action == "switch":
361	        if args.profile:
362	            ok, message = service.switch_profile(args.profile)
363	            return CommandResult(ok, message, 0 if ok else 1)
364	
365	        profiles = service.list_profiles()
366	        if not profiles:
367	            return CommandResult(False, "❌ 无可用配置", 1)
368	
369	        stdin_is_tty = bool(getattr(sys.stdin, "isatty", lambda: False)())
370	        stdout_is_tty = bool(getattr(sys.stdout, "isatty", lambda: False)())
371	        if not (stdin_is_tty and stdout_is_tty):
372	            lines = ["❌ 请指定 profile。可用配置："]
373	            for profile in profiles:
374	                lines.append(f"  - {profile['id']}")
375	            lines.append("示例：ai config switch <profile>")
376	            return CommandResult(False, "\n".join(lines), 2)
377	
378	        print("📋 可用配置：")
379	        for index, profile in enumerate(profiles, 1):
380	            current_mark = "⭐" if profile["current"] else "  "
381	            print(f"  {index}. {current_mark} {profile['id']} ({profile['name']})")
382	
383	        while True:
384	            try:
385	                choice = input("请输入序号或配置 ID（q 取消）: ").strip()
386	            except EOFError:
387	                return CommandResult(True, "✅ 已取消切换", 0)
388	            if not choice:
389	                print("请输入有效序号或配置 ID")
390	                continue
391	            lowered = choice.lower()
392	            if lowered in {"q", "quit", "exit"}:
393	                return CommandResult(True, "✅ 已取消切换", 0)
394	
395	            target = choice
396	            if choice.isdigit():
397	                index = int(choice) - 1
398	                if index < 0 or index >= len(profiles):
399	                    print("序号无效，请重新输入")
400	                    continue
401	                target = profiles[index]["id"]
402	
403	            ok, message = service.switch_profile(target)
404	            return CommandResult(ok, message, 0 if ok else 1)
405	    if args.action == "list":
406	        profiles = service.list_profiles()
407	        lines = ["📋 配置列表："]
408	        for profile in profiles:
409	            current_mark = "⭐" if profile["current"] else "  "
410	            stream_mark = "on" if profile["stream"] else "off"
411	            lines.append(f"  {current_mark} {profile['id']} ({profile['name']}) model={profile['model']} stream={stream_mark}")
412	        return CommandResult(True, "\n".join(lines))
413	    if args.action == "current":
414	        profile = service.get_active_profile()
415	        stream_mark = "on" if profile.stream else "off"
416	        message = (
417	            f"⭐ 当前配置：{profile.profile_id}\n"
418	            f"名称：{profile.name}\n模型：{profile.model}\nURL：{profile.api_url}\n流式：{stream_mark}"
419	        )
420	        return CommandResult(True, message)
421	    if args.action == "delete":
422	        ok, message = service.delete_profile(args.profile)
423	        return CommandResult(ok, message, 0 if ok else 1)
424	    if args.action == "stream":
425	        ok, message = service.set_stream(args.profile, args.enabled)
426	        return CommandResult(ok, message, 0 if ok else 1)
427	    if args.action == "export":
428	        ok, message = service.export_profile(args.profile, Path(args.output), args.redact)
429	        return CommandResult(ok, message, 0 if ok else 1)
430	    if args.action == "import":
431	        ok, message = service.import_profile(Path(args.input), args.profile)
432	        return CommandResult(ok, message, 0 if ok else 1)
433	    return CommandResult(False, "❌ 未知 config 子命令", 2)

447	def _dispatch(argv: list[str], ctx: AppContext) -> CommandResult:
448	    retry_aliases = {"重试", "再试", "再试一次", "retry", "try again", "继续"}
449	    if len(argv) == 1 and argv[0].strip().lower() in retry_aliases:
450	        ok, message = ctx.shell_service.run(argv[0].strip())
451	        return CommandResult(ok, message, 0 if ok else 1, data={"module_hint": "shell"})
452	
453	    if "--execute" in argv:
454	        return CommandResult(
455	            False,
456	            "❌ --execute 已下线，shell run 改为交互式分步执行：ai shell run \"查找大文件\"",
457	            2,
458	        )
459	
460	    if argv and argv[0].startswith("/"):
461	        return CommandResult(False, _migration_message(argv[0]), 2)
462	
463	    known_modules = {"chat", "file", "code", "context", "backup", "config", "shell"}
464	    if argv and argv[0] not in known_modules and not argv[0].startswith("-"):
465	        message = " ".join(argv).strip()
466	        output = ctx.chat_service.chat(message, use_history=True)
467	        stream_enabled = ctx.config_service.get_active_profile().stream
468	        suppress_print = bool(stream_enabled) and not output.startswith("❌")
469	        return _result_from_message(
470	            output,
471	            error_code=1,
472	            data={"streamed_output": suppress_print, "module_hint": "chat"},
473	        )
474	
475	    parser = _build_parser()
476	    try:
477	        args = parser.parse_args(argv)
478	    except ArgumentParsingExit as exc:
479	        output = exc.message.strip() if exc.message else parser.format_help().strip()
480	        return CommandResult(exc.exit_code == 0, output, exc.exit_code)
481	
482	    if args.clear:
483	        return _result_from_message(ctx.chat_service.clear_history())
484	
485	    if args.module is None:
486	        return CommandResult(True, parser.format_help().strip(), 0)
487	
488	    if args.module == "chat":
489	        message = " ".join(args.message).strip()
490	        if not message:
491	            return CommandResult(False, "❌ chat 需要消息文本，例如：ai chat 你好", 2)
492	        output = ctx.chat_service.chat(message, use_history=not args.no_history)
493	        stream_enabled = ctx.config_service.get_active_profile().stream
494	        suppress_print = bool(stream_enabled) and not output.startswith("❌")
495	        return _result_from_message(
496	            output,
497	            error_code=1,
498	            data={"streamed_output": suppress_print},
499	        )
500	
501	    handlers = {
502	        "file": _handle_file,
503	        "code": _handle_code,
504	        "context": _handle_context,
505	        "backup": _handle_backup,
506	        "config": _handle_config,
507	        "shell": _handle_shell,
508	    }
509	
510	    handler = handlers.get(args.module)
511	    if handler is None:
512	        return CommandResult(False, f"❌ 不支持的模块：{args.module}", 2)
513	    return handler(args, ctx)
514	
515	
516	def run(argv: list[str] | None = None) -> int:
517	    if argv is None:
518	        argv = sys.argv[1:]
519	
520	    ctx = AppContext()
521	    command_line = _format_command_line(argv)
522	
523	    try:
524	        result = _dispatch(argv, ctx)
525	    except KeyboardInterrupt:
526	        result = CommandResult(
527	            False,
528	            "❌ 操作已取消",
529	            130,
530	            data={"interrupted": True, "module_hint": (argv[0] if argv else "none")},
531	        )
532	    except Exception as exc:
533	        traceback_text = traceback.format_exc()
534	        result = CommandResult(False, f"❌ 内部错误：{exc}", 1, data={"traceback": traceback_text})
535	
536	    suppress_print = bool(result.data and result.data.get("streamed_output"))
537	    if result.message and not suppress_print:
538	        print(result.message)
539	
540	    history_output = result.message
541	    traceback_text = result.data.get("traceback") if result.data else None
542	    if traceback_text:
543	        history_output = f"{history_output}\n{traceback_text}".strip()
544	    try:
545	        ctx.history_service.append_command_record(
546	            command_line=command_line,
547	            output_text=history_output,
548	            ok=result.ok,
549	            exit_code=result.exit_code,
550	            metadata={
551	                "module": (result.data or {}).get("module_hint") or (argv[0] if argv else "none"),
552	                "interrupted": bool((result.data or {}).get("interrupted", False)),
553	            },
554	        )
555	    except Exception:
556	        pass
557	
558	    return result.exit_code
559	
560	
561	def main() -> None:
562	    raise SystemExit(run())

566	    main()
